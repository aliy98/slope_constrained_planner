172 inline void setSO3FromRPY(ob::SO3StateSpace::StateType& s, double* rpy) {
173  const auto r2 = rpy[0]*0.5;
174  const auto p2 = rpy[1]*0.5;
175  const auto y2 = rpy[2]*0.5;
176  const auto cr = cos(r2);
177  const auto cp = cos(p2);
178  const auto cy = cos(y2);
179  const auto sr = sin(r2);
180  const auto sp = sin(p2);
181  const auto sy = sin(y2);
182  s.w = cy * cp * cr + sy * sp * sr;
183  s.x = cy * cp * sr - sy * sp * cr;
184  s.y = sy * cp * sr + cy * sp * cr;
185  s.z = sy * cp * cr - cy * sp * sr;
186 }
